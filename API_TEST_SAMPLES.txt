# ======================================================
# 1. AUTH APIs (Authentication)
# ======================================================

# --- Register User ---
POST /api/auth/register
Content-Type: application/json
Body:
{
  "email": "user@example.com",
  "password": "123456",
  "fullName": "Nguyễn Văn A",
  "phone": "0987654321"
}

# --- Login ---
POST /api/auth/login
Content-Type: application/json
Body:
{
  "email": "user@example.com",
  "password": "123456"
}

# --- Get My Profile ---
GET /api/auth/me
Cookie: JWT_TOKEN=<token>

# --- Update My Profile ---
PUT /api/auth/me
Content-Type: application/json
Cookie: JWT_TOKEN=<token>
Body:
{
  "fullName": "Nguyễn Văn B",
  "phone": "0987654322",
  "address": "TP.HCM"
}

# --- Refresh Token ---
POST /api/auth/refresh
Content-Type: application/json
Body:
{
  "refreshToken": "<refresh_token>"
}

# --- Logout ---
POST /api/auth/logout
Content-Type: application/json
Cookie: JWT_TOKEN=<token>
Body:
{
  "refreshToken": "<refresh_token>"
}

# --- Forgot Password ---
POST /api/auth/forgot-password
Content-Type: application/json
Body:
{
  "email": "user@example.com"
}

# --- Verify Email ---
POST /api/auth/verify-email
Content-Type: application/json
Body:
"<token>"

# --- Resend Verification Email ---
POST /api/auth/resend-verification-email
Content-Type: application/json
Body:
"user@example.com"

# ======================================================
# 2. USER APIs (Public - Books & Categories)
# ======================================================

# --- Get All Categories ---
GET /api/users/categories

# --- Search Books ---
GET /api/users/books?keyword=java&page=0&size=10

# --- Get Book Detail ---
GET /api/users/books/{id}


# ======================================================
# 3. COMMENT APIs (User - requires auth)
# ======================================================

# --- Get Comments by Book ---
GET /api/book/{bookId}/comments

# --- Create Comment ---
POST /api/book/{bookId}/comments
Content-Type: application/json
Cookie: JWT_TOKEN=<token>
Body:
{
  "content": "Sách rất hay, đáng đọc!"
}

# --- Delete Comment ---
DELETE /api/book/{bookId}/comments/{commentId}
Cookie: JWT_TOKEN=<token>


# ======================================================
# 4. CART APIs (User - requires auth)
# ======================================================

# --- Get My Cart ---
GET /api/users/cart
Cookie: JWT_TOKEN=<token>

# --- Add to Cart ---
POST /api/users/cart/add?bookId=1&quantity=2
Cookie: JWT_TOKEN=<token>

# --- Update Cart Quantity ---
PUT /api/users/cart/update?bookId=1&quantity=3
Cookie: JWT_TOKEN=<token>

# --- Remove from Cart ---
DELETE /api/users/cart/remove?bookId=1
Cookie: JWT_TOKEN=<token>

# --- Clear Cart ---
DELETE /api/users/cart/clear
Cookie: JWT_TOKEN=<token>

# --- Checkout ---
POST /api/users/cart/checkout
Cookie: JWT_TOKEN=<token>


# ======================================================
# 5. ORDER APIs (User - requires auth)
# ======================================================

# --- Get Order By ID ---
